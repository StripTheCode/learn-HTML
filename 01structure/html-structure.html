Basic structure of a HTML page.

        
            

                // HTML Structure
                <!DOCTYPE html>

                // HTML Tag
                <html lang="en">

                // Head Section
                <head>
                    // Meta tags
                    <meta charset="UTF-8">
                    <meta name="viewport" content="width=device-width, initial-scale=1.0">
                    // Title
                    <title>HTML Structure</title>
                // End of Head Section
                </head>

                // Body Section
                <body>

                    // Content
                    <!-- WRITE YOUR CODE HERE -->

                    <h1>Hello World!</h1>

                // End of Body
                </body>

                // End of HTML
                </html>
